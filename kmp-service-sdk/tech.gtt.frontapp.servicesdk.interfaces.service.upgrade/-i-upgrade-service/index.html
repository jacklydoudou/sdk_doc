IUpgradeService

expect interface IUpgradeService : IService

Interface for managing application upgrade operations.

This service provides functionality for checking available upgrades, initiating the upgrade process, and monitoring upgrade events. It handles both regular application updates and Docker container upgrades.

Example usage:

// Get upgrade service instance
val upgradeService = serviceSDK.getService(IUpgradeService::class)

// Check for available upgrades
upgradeService.checkUpgrade()
    .proceed(object : SDKRequestCallback<List<IAppUpgradeInfo>> {
        override fun onSuccess(upgrades: List<IAppUpgradeInfo>?) {
            upgrades?.forEach { upgrade ->
                println("New version: ${upgrade.versionName}")
                println("Size: ${upgrade.size} bytes")
                println("Type: ${upgrade.upgradeType}")
            }
        }
        override fun onFail(exception: SDKException) {
            println("Failed to check upgrades")
        }
    })

// Monitor upgrade events
val dispose = upgradeService.registerUpgradeEvent { event ->
    when (event.type) {
        UpgradeEventType.UPGRADE_UPDATE_EVENT -> {
            println("Docker upgrade event received")
        }
    }
}

// Start upgrade process
upgradeService.startUpgrade()
    .proceed(object : SDKRequestCallback<Unit> {
        override fun onSuccess(unit: Unit?) {
            println("Upgrade started successfully")
        }
        override fun onFail(exception: SDKException) {
            println("Failed to start upgrade")
        }
    })
