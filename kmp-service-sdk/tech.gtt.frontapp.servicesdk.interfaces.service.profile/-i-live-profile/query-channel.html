expect abstract fun queryChannel(queryChannelParam: QueryChannelParam): SDKRequest<List<ProfileChannel>>

Retrieves channel configurations based on specified criteria.

This method allows querying channels from either viewing history or favorites list. The operation is asynchronous and returns a SDKRequest that must be executed using proceed() with a callback.

Example:

// Query favorite channels
liveProfile.queryChannel(QueryChannelParam.favorite())
    .proceed(object : SDKRequestCallback<List<ProfileChannel>> {
        override fun onSuccess(channels: List<ProfileChannel>?) {
            channels?.forEach { channel ->
                println("Channel ID: ${channel.logicalChannelId}")
            }
        }
        override fun onFail(exception: SDKException) {
            println("Failed to query channels")
        }
    })
