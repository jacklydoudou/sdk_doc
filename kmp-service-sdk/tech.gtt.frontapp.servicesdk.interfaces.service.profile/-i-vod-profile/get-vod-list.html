expect abstract fun getVodList(queryVodParam: QueryVodProfileParam, forceSync: Boolean, pageNo: Int, pageSize: Int): SDKRequest<List<IProfileVodInfo>>

Retrieves a paginated list of VOD entries.

This method fetches VOD entries (history or favorites) with support for pagination and server synchronization. The operation is asynchronous and returns a SDKRequest that must be executed using proceed() with a callback.

Example:

vodProfile.getVodList(
    queryVodParam = QueryVodProfileParam.companion.history(),
    forceSync = true,
    pageNo = 1,
    pageSize = 20
).proceed(object : SDKRequestCallback<List<IProfileVodInfo>> {
    override fun onSuccess(content: List<IProfileVodInfo>?) {
        content?.forEach { vod ->
            println("Title: ${vod.title}")
        }
    }
    override fun onFail(exception: SDKException) {
        println("Failed to get content list")
    }
})
